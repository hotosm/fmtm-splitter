30 def create_connection(
31  db: Union[str, psycopg2.extensions.connection],
32 ) -> psycopg2.extensions.connection:
33  """Get db connection from existing psycopg2 connection, or URL string.
34 
35  Args:
36  db (str, psycopg2.extensions.connection):
37  string or existing db connection.
38  If `db` is a string, a new connection is generated.
39  If `db` is a psycopg2 connection, the connection is re-used.
40 
41  Returns:
42  conn: DBAPI connection object to generate cursors from.
43  """
44  # Makes Postgres UUID, JSONB usable, else error
45  register_uuid()
46  register_adapter(dict, Json)
47 
48  if isinstance(db, psycopg2.extensions.connection):
49  conn = db
50  elif isinstance(db, str):
51  conn = psycopg2.connect(db)
52  else:
53  msg = "The `db` variable is not a valid string or psycopg2 connection."
54  log.error(msg)
55  raise ValueError(msg)
56 
57  return conn
